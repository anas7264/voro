// 02 — Scroll Reveal System

FIVE
Reveals

Each card uses a different entrance animation — triggered when it enters the viewport. Scroll past this section and back to replay.

fadeUp
Fade Up
translateY(60px) + opacity. The workhorse. Use for cards, text blocks, any vertical entry.
slideLeft
Slide Left
translateX(80px) + opacity. For horizontal panels, sidebar items entering from the right edge.
scaleIn
Scale In
scale(0.85) + opacity with spring easing. For feature cards that need presence and weight.
clipReveal
Clip Reveal
clip-path: inset(0 100% 0 0) → 0%. Cinematic wipe from left. Best for headlines and images.
blurIn
Blur In
blur(20px) + opacity with silk easing. For ambient reveals, background elements coming into focus.

/* ══ VORO Motion Design System — Chapter 05 ══ */ :root { /* ── Easing library ──────────────────────────────────────── */ --ease-spring: cubic-bezier(0.34, 1.56, 0.64, 1 ); /* Elastic overshoot */ --ease-expo-out: cubic-bezier(0.16, 1, 0.3, 1 ); /* Fast → settles */ --ease-expo-in: cubic-bezier(0.7, 0, 0.84, 0 ); /* Accelerates out */ --ease-silk: cubic-bezier(0.25, 0.46, 0.45, 0.94); /* Smooth luxury */ --ease-snap: cubic-bezier(0.35, 0, 0.1, 1 ); /* UI crisp, precise */ --ease-back: cubic-bezier(0.68, -0.55, 0.265,1.55); /* Wind-up overshoot */ } /* ── Scroll reveals ──────────────────────────────────────── */ [data-reveal="fade-up"] { opacity:0; transform:translateY(60px); } [data-reveal="slide-left"] { opacity:0; transform:translateX(80px); } [data-reveal="scale-in"] { opacity:0; transform:scale(0.85); } [data-reveal="clip"] { clip-path:inset(0 100% 0 0); } [data-reveal="blur-in"] { opacity:0; filter:blur(20px); } [data-reveal].revealed { opacity:1; transform:none; clip-path:inset(0 0% 0 0); filter:blur(0); transition: all 0.75s var(--ease-expo-out); transition-delay: calc(var(--i, 0) * 60ms); } /* ── Magnetic button ─────────────────────────────────────── */ .btn-magnetic { transition: transform 0.3s var(--ease-expo-out); will-change: transform; } /* ── 3D Tilt card ────────────────────────────────────────── */ .card-tilt { transform: perspective(1000px) rotateX(var(--rx,0deg)) rotateY(var(--ry,0deg)) translateZ(var(--tz,0px)); transition: transform 0.15s ease; } /* ── Stagger children ────────────────────────────────────── */ [data-stagger] > * { transition-delay: calc(var(--i) * 60ms); }
